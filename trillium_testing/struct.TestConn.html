pub async fn run_async(self, handler: &impl Handler) -> Self

runs this conn against a handler and finalizes response headers, asynchronously. Since most tests are performed in a sync context, most of the time it is preferable to use TestConn::run, also aliased as TestConn::on. This function is aliased as TestConn::async_on.

use trillium_testing::prelude::*;

trillium_testing::block_on(async {
    async fn handler(conn: Conn) -> Conn {
        conn.ok("hello trillium")
    }

    let conn = get("/").run_async(&handler).await;
    assert_ok!(conn, "hello trillium", "content-length" => "14");
});
